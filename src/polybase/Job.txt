@public
collection Job {
  id: string; // Bacalhau Job ID
  user: string;
  status: string;
  timestamp:number;
  
  @index(user, timestamp);
  
  constructor(id: string, user: string) {
    this.id = id;
    this.user = user;
    this.status = "created";
    this.timestamp = Date.now();
  }

  updateStatus(status: string) {
    this.status = status;
  }
}
